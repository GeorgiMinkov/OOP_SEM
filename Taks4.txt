#include <iostream.h>
// Създава се клас K
class K
{// Член данни а - целочислен тип; b - реален тип (с плаваща запетая)
private:
    int a; 
    double b;
//Член функции
public:
    K(int, double); // Конструктор по подадени параметри
    K(K const &); // Конструктор по подаден обект
    void display() const; // Член функция за достъп
};
// Задава на a и b стойностите подадени в x и y
K::K(int x, double y)
{
    a = x;
    b = y;
}
// Задава на a и b стойностите на член данните в обекта r
K::K(K const &r)
{
    a = r.a;
    b = r.b;
}
// Извежда на екране стойностите на a и b
void K::display() const
{
    cout << a << " " << b << endl;
}

int main()
{
    K e1(4, 1.8), *e2 = &e1; // Създава обекта е1 от тип K със стойности: a = 4; b = 1.8. Създаване на е2 - указател от K, който приема адреса на е1
    e1.display(); // Извежда на екрана: 4 1.8
    e2->display(); // Извежда на екрана: 4 1.8
    K e3 = *e2; // Създава обект е3 от тип К, като приема стойностите на *е2, т.е на е1, т.е а = 4 и b = 1.8
    e3.display(); // Извежда на екрана: 4 1.8
    return 0;
}
