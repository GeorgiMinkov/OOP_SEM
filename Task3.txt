#include <iostream.h>
//Създаване на класа H
class H
{
private:
    int a, b; // член данни a и b от целочислен тип int. 

public:
    H(int = 1, int = 1); // Конструктор по подразбиране, както и конструктор по подадени аргументи.
    H(H const &); // Конструктор по подаден аргумент от тип H
    void display() const; // Член функция за достъп
};
// Подава на a и b стойностите зададени в x и y
H::H(int x, int y)
{
    a = x;
    b = y;
}
// Подава на a и b стойностите зададени в обекта r 
H::H(H const &r)
{
    a = r.a;
    b = r.b;
}
// Изход на екрана на стойносттите на а и b
void H::display() const
{
    cout << a << " " << b << endl;
}

int main()
{
    H e1, e2(6), e3(3, 8); // Създава обекти от тип H като на е1 се свързва конструктотър по подразбиране(т.е a = 1 и b = 1), е2 -> a = 6 и b = 1, e3 -> a = 3 и b = 8
    //Извежда съответните стойности на обектите е1, е2 и е3
    e1.display(); // 1 1
    e2.display(); // 6 1
    e3.display(); // 3 8

    H e4 = e3; // Създава се обект е4 от тип H, който се инизиализира със стойностите на е3 (а = 3 и  b = 8)

    e4.display(); // Извежда: 3 8 
    return 0;
}
